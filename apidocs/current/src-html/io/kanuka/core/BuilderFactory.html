001package io.kanuka.core;
002
003import java.util.List;
004
005/**
006 * Factory for creating Builder instances.
007 *
008 * These Builders are typically used by generated code (Java annotation processing - kanuka-generator).
009 */
010public class BuilderFactory {
011
012  /**
013   * Create the root level Builder.
014   *
015   * @param suppliedBeans The list of beans (typically test doubles) supplied when building the context.
016   */
017  public static Builder newRootBuilder(List<Object> suppliedBeans) {
018    return new DBuilder(suppliedBeans);
019  }
020
021  /**
022   * Create a Builder for the named context (module).
023   *
024   * @param name the name of the module / bean context
025   * @param dependsOn the names of modules this module is depends on.
026   */
027  public static Builder newBuilder(String name, String... dependsOn) {
028    return new DBuilder(name, dependsOn);
029  }
030}
